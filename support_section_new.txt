                  <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
                    <div className="lg:col-span-2 bg-white border border-gray-100 rounded-3xl p-6 shadow-sm space-y-4">
                      <div className="flex items-center justify-between">
                        <div>
                          <p className="text-xs font-semibold text-gray-500 uppercase tracking-wide">Recent tickets</p>
                          <h3 className="text-xl font-bold text-gray-900">Track your requests</h3>
                        </div>
                        <button onClick={fetchSupport} className="text-sm font-semibold text-gray-600 hover:text-orange-600">
                          {isSupportLoading ? 'Refreshing...' : 'Refresh'}
                        </button>
                      </div>

                      {isSupportLoading && !supportData && (
                        <div className="flex items-center justify-center py-8">
                          <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-orange-600"></div>
                        </div>
                      )}

                      {supportError && (
                        <div className="bg-rose-50 border border-rose-200 rounded-2xl p-4 text-sm text-rose-700">
                          {supportError}
                        </div>
                      )}

                      {!isSupportLoading && !supportError && supportData && (
                        <>
                          {supportData.tickets.length === 0 ? (
                            <div className="py-8 text-center">
                              <p className="text-gray-500 text-sm">No support tickets yet.</p>
                              <p className="text-gray-400 text-xs mt-1">Open a ticket to get help from our support team.</p>
                            </div>
                          ) : (
                            <div className="space-y-3">
                              {supportData.tickets.slice(0, 3).map((ticket) => {
                                const statusColors: Record<string, string> = {
                                  OPEN: 'bg-amber-50 text-amber-700 border-amber-100',
                                  IN_PROGRESS: 'bg-blue-50 text-blue-700 border-blue-100',
                                  WAITING_USER: 'bg-purple-50 text-purple-700 border-purple-100',
                                  RESOLVED: 'bg-green-50 text-green-700 border-green-100',
                                  CLOSED: 'bg-gray-50 text-gray-700 border-gray-100'
                                }
                                const timeAgo = (dateString: string) => {
                                  const date = new Date(dateString)
                                  const now = new Date()
                                  const diffMs = now.getTime() - date.getTime()
                                  const diffMins = Math.floor(diffMs / 60000)
                                  if (diffMins < 60) return `${diffMins}m ago`
                                  const diffHours = Math.floor(diffMins / 60)
                                  if (diffHours < 24) return `${diffHours}h ago`
                                  const diffDays = Math.floor(diffHours / 24)
                                  return `${diffDays}d ago`
                                }
                                return (
                                  <div key={ticket.id} className="flex items-center justify-between border border-gray-100 rounded-2xl px-4 py-3 hover:bg-gray-50 transition-colors cursor-pointer">
                                    <div className="flex-1">
                                      <p className="text-sm font-semibold text-gray-900">{ticket.subject}</p>
                                      <div className="flex items-center gap-2 mt-1">
                                        <p className="text-xs text-gray-500">{ticket.category}</p>
                                        <span className="text-gray-300">•</span>
                                        <p className="text-xs text-gray-500">Updated {timeAgo(ticket.updatedAt)}</p>
                                        {ticket.messageCount > 0 && (
                                          <>
                                            <span className="text-gray-300">•</span>
                                            <p className="text-xs text-gray-500">{ticket.messageCount} {ticket.messageCount === 1 ? 'message' : 'messages'}</p>
                                          </>
                                        )}
                                      </div>
                                    </div>
                                    <span className={`px-3 py-1 rounded-full text-xs font-bold border ${statusColors[ticket.status] || 'bg-gray-50 text-gray-700 border-gray-100'}`}>
                                      {ticket.status.replace('_', ' ')}
                                    </span>
                                  </div>
                                )
                              })}
                            </div>
                          )}
                        </>
                      )}

                      <div className="flex items-center gap-3 pt-2">
                        <button className="px-4 py-3 bg-orange-600 text-white rounded-xl text-sm font-bold hover:bg-orange-700 transition-colors">
                          Submit new request
                        </button>
                        <button className="text-sm font-semibold text-gray-700 hover:text-orange-600">
                          Message support
                        </button>
                      </div>
                    </div>
